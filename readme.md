```js
npm install
npm link

// register current folder with alias 'work-space'
mom add my-workspace
mom -a my-workspace
mom --add my-workspace

// remove the alias
mom delete my-workspace
mom -d my-workspace
mom --delete my-workspace


// open it with 'code'
mom open my-workspace
mom -o my-workspace
mom --open my-workspace

// show all your aliases
mom -l
mom list
```
